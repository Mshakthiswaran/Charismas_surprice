            const setupCanvas = (id) => {
                const c = document.getElementById(id);
                const ctx = c.getContext('2d');
                c.width = window.innerWidth;
                c.height = window.innerHeight;
                return { c, ctx, w: c.width, h: c.height };
            }

            const snow = setupCanvas('snowCanvas');
            const flakes = Array.from({ length: 150 }, () => ({
                x: Math.random() * snow.w,
                y: Math.random() * snow.h,
                r: Math.random() * 3 + 1,
                s: Math.random() * 1 + 0.5
            }));

            function drawSnow() {
                snow.ctx.clearRect(0, 0, snow.w, snow.h);
                snow.ctx.fillStyle = "rgba(255,255,255,0.8)";
                snow.ctx.beginPath();
                flakes.forEach(f => {
                    snow.ctx.moveTo(f.x, f.y);
                    snow.ctx.arc(f.x, f.y, f.r, 0, Math.PI * 2);
                    f.y += f.s;
                    if (f.y > snow.h) f.y = 0;
                });
                snow.ctx.fill();
                requestAnimationFrame(drawSnow);
            }
            drawSnow();

            const pC = document.getElementById('particlesCanvas');
            const pCtx = pC.getContext('2d');
            pC.width = window.innerWidth;
            pC.height = window.innerHeight;
            let particles = [];
            function createBurst(x, y) {
                for (let i = 0; i < 20; i++) {
                    particles.push({
                        x,
                        y,
                        vx: Math.random() * 4 - 2,
                        vy: Math.random() * 4 - 2,
                        a: 1
                    });
                }
            }

            function drawParticles() {
                pCtx.clearRect(0, 0, pC.width, pC.height);
                particles.forEach((p, i) => {
                    p.x += p.vx;
                    p.y += p.vy;
                    p.a -= 0.02;
                    pCtx.fillStyle = `rgba(212,175,55,${p.a})`;
                    pCtx.fillRect(p.x, p.y, 3, 3);
                    if (p.a <= 0) particles.splice(i, 1);
                });
                requestAnimationFrame(drawParticles);
            }
            drawParticles();
            // Handle window resize
            window.addEventListener('resize', () => {

            function typeWriter(element, text, speed = 20) {
                return new Promise((resolve) => {

                    element.textContent = '';
                    element.classList.add('typing');

                    function type() {
                        if (i < text.length) {
                            element.textContent += text.charAt(i);
                            i++;
                            setTimeout(type, speed);
                        } else {
                            resolve();
                        }
                    }
                    type();
                });
            }

const parkSanta = () => {
    if (santaElement) {
        // Clear any previous inline transform (from the attempt to freeze position)
        santaElement.style.transform = ''; 
        
        // Add the 'parked' class, which stops the animation and moves him to the corner
        santaElement.classList.add('parked'); 
    }
}

const resumeFlying = () => {
    if (santaElement) {
        // Remove the 'parked' class. This automatically re-enables the
        // 'FlyingSanta' animation, which will start from the current (0%, 0%)
        // position and continue flying across the screen.
        santaElement.classList.remove('parked');
    }
    modal.classList.remove('active');
    modalVideo.pause();
}

});
